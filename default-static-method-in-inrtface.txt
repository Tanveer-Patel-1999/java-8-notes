package com.example.defaultstatic;

interface Vehicle{

    String getBrand();

    String speedUp();

    String slowDown();

    //    we dose not required to override default and static method
    default String turnAlarmOn()
    {
        return "turning vehicle alarm on";
    }

    default String turnAlarmOf()
    {
        return "turning vehicle alarm of";
    }

    static String getCompany()
    {
        return "BWM";
    }

}
//this class can implement the vehicle interface
class Car implements Vehicle
{

    @Override
    public String getBrand() {
        return "BMW";
    }

    @Override
    public String speedUp() {
        return "the car is speeding up";
    }

    @Override
    public String slowDown() {
        return "the car is slowing down";
    }
}

public class DefaultStaticDemo {
    public static void main(String[] args) {
        //create a ref variable  of interface
        Vehicle vehicle = new Car();
        System.out.println(vehicle.getBrand());
        System.out.println(vehicle.speedUp());
        System.out.println(vehicle.slowDown());

        System.out.println(vehicle.turnAlarmOf());
        System.out.println(vehicle.turnAlarmOn());

        //we can call static method using interface
        System.out.println(Vehicle.getCompany());

    }
}
